# 4. Zsh Completion (kubectl + helm)

```code
if ! grep -q "kubectl completion zsh" ~/.zshrc; then
  echo '' >> ~/.zshrc
  echo '# kubectl completion' >> ~/.zshrc
  echo 'source <(kubectl completion zsh)' >> ~/.zshrc
  echo 'alias k=kubectl' >> ~/.zshrc
fi

if ! grep -q "helm completion zsh" ~/.zshrc; then
  echo '' >> ~/.zshrc
  echo '# helm completion' >> ~/.zshrc
  echo 'source <(helm completion zsh)' >> ~/.zshrc
fi
```
